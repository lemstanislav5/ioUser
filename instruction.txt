CREATE TABLE chat_db.users (
	id INT auto_increment NOT NULL,
	domain_or_ip varchar(100) NOT NULL,
	chat_id_telegram varchar(100) NOT NULL,
	first_name varchar(100) NOT NULL,
	last_name varchar(100) NOT NULL,
	registration_date varchar(100) NOT NULL,
	finance varchar(100) NOT NULL,
	ban varchar(100) NOT NULL,
  PRIMARY KEY (id)
)
DEFAULT CHARSET=utf8
COLLATE=utf8_general_ci;

CREATE TABLE chat_db.messages (
  id INT auto_increment NOT NULL,
  id_message varchar(100) NOT NULL,
  date varchar(100) NOT NULL,
  domain_or_ip varchar(100) NOT NULL,
  customer varchar(100) NOT NULL,
  message varchar(100) NOT NULL,
  PRIMARY KEY (id)
)
DEFAULT CHARSET=utf8
COLLATE=utf8_general_ci;
